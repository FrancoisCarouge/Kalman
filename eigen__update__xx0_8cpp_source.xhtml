57template <std::size_t StateSize, std::size_t OutputSize>
58void eigen_update(::benchmark::State &state) {
59
60 using kalman = kalman<vector<float, StateSize>, vector<float, OutputSize>>;
61
62 kalman filter;
63 std::random_device random_device;
64 std::mt19937 random_generator(random_device());
65 std::uniform_real_distribution uniformly_distributed(0.f, 1.f);
66
67 for (auto _ : state) {
68
69 typename kalman::output z;
70
71 fcarouge::internal::for_constexpr<std::size_t{0}, OutputSize, 1>(
72 [&z, &uniformly_distributed, &random_generator](auto position) {
73 z[position] = uniformly_distributed(random_generator);
74 });
75
76 ::benchmark::ClobberMemory();
77 const auto start{clock::now()};
78
79 filter.update(z);
80
81 ::benchmark::ClobberMemory();
82 const auto end{clock::now()};
83
84 state.SetIterationTime(
85 std::chrono::duration_cast<std::chrono::duration<double>>(end - start)
86 .count());
87 }
88}
