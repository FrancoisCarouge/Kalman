66[[maybe_unused]] auto sample{[] {
67 // A one-dimensional filter, constant system dynamic model.
68 kalman filter{
69 // We initialize the Kalman filter and predict the next state (which is
70 // the first state). We don't know what the temperature of the liquid is,
71 // and our guess is: estimated state X = 10°C.
72 state{10.},
73 // The measured liquid temperature Z.
74 output<double>,
75 // Our guess is very imprecise, so we set our initialization estimate
76 // error σ to 100. The estimate uncertainty p of the initialization is the
77 // error variance σ^2: P = p0,0 = 100^2 = 10,000. This variance is very
78 // high. If we initialize with a more meaningful value, we will get faster
79 // Kalman filter convergence.
80 estimate_uncertainty{10'000.},
81 // We have an accurate model, thus we set the process uncertainty noise
82 // variance Q to 0.0001.
83 process_uncertainty{0.000'1},
84 // Since the measurement error of the thermometer is σ = 0.1, the variance
85 // σ^2 would be 0.01, thus the measurement, output uncertainty is: R = r1
86 // = 0.01. The measurement error (standard deviation) is 0.1 degrees
87 // Celsius.
88 output_uncertainty{0.01}};
89
90 // Now, we shall predict the next state based on the initialization values.
91 filter.predict();
92
93 assert(10 == filter.x() &&
94 "Since our model has constant dynamics, the predicted estimate is "
95 "equal to the current estimate: x^1,0 = 10°C.");
96 assert(10'000.000'1 == filter.p() &&
97 "The extrapolated estimate uncertainty (variance): p1,0 = p0,0 + q = "
98 "10000 + 0.0001 = 10000.0001.");
99
100 // The first measurement value: z1 = 49.95°C. Measure and update.
101 filter.update(49.95);
102
103 assert(std::abs(1 - filter.k() / 0.999'999) < 0.0001 &&
104 "The gain expected at 0.01% accuracy.");
105
106 // And so on, run a step of the filter, predicting and updating, every
107 // measurements period: Δt = 5s (constant).
108 const auto step{[&filter](double temperature) {
109 filter.predict();
110 filter.update(temperature);
111 }};
112
113 step(49.967);
114 step(50.1);
115 step(50.106);
116 step(49.992);
117 step(49.819);
118 step(49.933);
119 step(50.007);
120 step(50.023);
121 step(49.99);
122
123 // The estimate uncertainty quickly goes down, after 10 measurements:
124 assert(std::abs(1 - filter.p() / 0.001'3) < 0.05 &&
125 "The estimate uncertainty expected at 5% accuracy."
126 "The estimate uncertainty is 0.0013, i.e. the estimate error standard "
127 "deviation is: 0.036°C.");
128 assert(std::abs(1 - filter.x() / 49.988) < 0.001 &&
129 "The state estimates expected at 0.1% accuracy."
130 "The filter estimates the liquid temperature at 49.988°C.");
131 assert(std::abs(1 - filter.k() / 0.126'5) < 0.001 &&
132 "The gain expected at 0.1% accuracy.");
133
134 // So we can say that the liquid temperature estimate is: 49.988 ± 0.036°C.
135 // In this example we've measured a liquid temperature using the
136 // one-dimensional Kalman filter. Although the system dynamics include a
137 // random process noise, the Kalman filter can provide good estimation.
138
139 return 0;
140}()};
