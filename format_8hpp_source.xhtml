51template <typename State, typename Output, typename Input, typename UpdateTypes,
52 typename PredictionTypes, typename Char>
53// It is allowed to add template specializations for any standard library class
54// template to the namespace std only if the declaration depends on at least one
55// program-defined type and the specialization satisfies all requirements for
56// the original template, except where such specializations are prohibited.
57// NOLINTNEXTLINE(cert-dcl58-cpp)
58struct std::formatter<
59 fcarouge::kalman<State, Output, Input, UpdateTypes, PredictionTypes>,
60 Char> {
61 constexpr auto parse(std::basic_format_parse_context<Char> &parse_context) {
62 return parse_context.begin();
63 }

66 template <typename OutputIt>
67 auto format(const fcarouge::kalman<State, Output, Input, UpdateTypes,
68 PredictionTypes> &filter,
69 std::basic_format_context<OutputIt, Char> &format_context) const
70 -> OutputIt {
71 format_context.advance_to(
72 format_to(format_context.out(), R"({{"f": {}, )", filter.f()));
73
74 if constexpr (requires { filter.g(); }) {
75 format_context.advance_to(
76 format_to(format_context.out(), R"("g": {}, )", filter.g()));
77 }
78
79 format_context.advance_to(format_to(format_context.out(),
80 R"("h": {}, "k": {}, "p": {}, )",
81 filter.h(), filter.k(), filter.p()));
82
83 {
84 constexpr auto end{fcarouge::internal::repack_s<PredictionTypes>};
85 constexpr decltype(end) begin{0};
86 constexpr decltype(end) next{1};
87 fcarouge::internal::for_constexpr<begin, end, next>(
88 [&format_context, &filter](auto position) {
89 format_context.advance_to(
90 format_to(format_context.out(), R"("prediction_{}": {}, )",
91 position(), filter.template predict<position>()));
92 });
93 }
94
95 format_context.advance_to(format_to(format_context.out(),
96 R"("q": {}, "r": {}, "s": {}, )",
97 filter.q(), filter.r(), filter.s()));
98
99 if constexpr (requires { filter.u(); }) {
100 format_context.advance_to(
101 format_to(format_context.out(), R"("u": {}, )", filter.u()));
102 }
103
104 {
105 constexpr auto end{fcarouge::internal::repack_s<UpdateTypes>};
106 constexpr decltype(end) begin{0};
107 constexpr decltype(end) next{1};
108 fcarouge::internal::for_constexpr<begin, end, next>(
109 [&format_context, &filter](auto position) {
110 format_context.advance_to(
111 format_to(format_context.out(), R"("update_{}": {}, )",
112 position(), filter.template update<position>()));
113 });
114 }
115
116 format_context.advance_to(format_to(format_context.out(),
117 R"("x": {}, "y": {}, "z": {}}})",
118 filter.x(), filter.y(), filter.z()));
119
120 return format_context.out();
121 }
122};
