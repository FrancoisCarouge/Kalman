52template <fcarouge::kalman_internal::kalman_filter Filter, typename Char>
53// It is allowed to add template specializations for any standard library class

58struct std::formatter<Filter, Char> {
59 constexpr auto parse(std::basic_format_parse_context<Char> &parse_context) {
60 return parse_context.begin();
61 }

64 template <typename FormatContext>
65 constexpr auto
66 format(const Filter &filter,
67 FormatContext &format_context) const -> FormatContext::iterator {
68 format_context.advance_to(std::format_to(format_context.out(), R"({{)"));
69
70 if constexpr (fcarouge::kalman_internal::has_state_transition_method<
71 Filter>) {
72 format_context.advance_to(
73 std::format_to(format_context.out(), R"("f": {}, )", filter.f()));
74 }
75
76 if constexpr (fcarouge::kalman_internal::has_input_control_method<Filter>) {
77 format_context.advance_to(
78 std::format_to(format_context.out(), R"("g": {}, )", filter.g()));
79 }
80
81 if constexpr (fcarouge::kalman_internal::has_output_model_method<Filter>) {
82 format_context.advance_to(
83 std::format_to(format_context.out(), R"("h": {}, )", filter.h()));
84 }
85
86 format_context.advance_to(std::format_to(
87 format_context.out(), R"("k": {}, "p": {}, )", filter.k(), filter.p()));
88
89 if constexpr (fcarouge::kalman_internal::has_prediction_types<Filter>) {
90 fcarouge::kalman_internal::for_constexpr<
91 0, fcarouge::kalman_internal::size<typename Filter::prediction_types>,
92 1>([&format_context, &filter](auto position) {
93 format_context.advance_to(
94 std::format_to(format_context.out(), R"("prediction_{}": {}, )",
95 position(), filter.template predict<position>()));
96 });
97 }
98
99 if constexpr (fcarouge::kalman_internal::has_process_uncertainty_method<
100 Filter>) {
101 format_context.advance_to(
102 std::format_to(format_context.out(), R"("q": {}, )", filter.q()));
103 }
104
105 if constexpr (fcarouge::kalman_internal::has_output_uncertainty<Filter>) {
106 format_context.advance_to(
107 std::format_to(format_context.out(), R"("r": {}, )", filter.r()));
108 }
109
110 format_context.advance_to(
111 std::format_to(format_context.out(), R"("s": {}, )", filter.s()));
112
115 if constexpr (fcarouge::kalman_internal::has_input_method<Filter>) {
116 format_context.advance_to(
117 std::format_to(format_context.out(), R"("u": {}, )", filter.u()));
118 }
119
121 if constexpr (fcarouge::kalman_internal::has_update_types<Filter>) {
122 fcarouge::kalman_internal::for_constexpr<
123 0, fcarouge::kalman_internal::size<typename Filter::update_types>, 1>(
124 [&format_context, &filter](auto position) {
125 format_context.advance_to(
126 std::format_to(format_context.out(), R"("update_{}": {}, )",
127 position(), filter.template update<position>()));
128 });
129 }
130
131 format_context.advance_to(
132 std::format_to(format_context.out(), R"("x": {}, "y": {}, "z": {}}})",
133 filter.x(), filter.y(), filter.z()));
134
135 return format_context.out();
136 }
137};
