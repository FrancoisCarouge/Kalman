51void bench(::benchmark::State &state) {
52 using kalman = kalman<float, float, float>;
53
54 kalman filter;
55 std::random_device random_device;
56 std::mt19937 random_generator{random_device()};
57 std::uniform_real_distribution<float> uniformly_distributed;
58
59 for (auto _ : state) {
60 const typename kalman::output z{uniformly_distributed(random_generator)};
61
62 ::benchmark::ClobberMemory();
63 const auto start{clock::now()};
64
65 filter.update(z);
66
67 ::benchmark::ClobberMemory();
68 const auto end{clock::now()};
69
70 state.SetIterationTime(std::chrono::duration<double>{end - start}.count());
71 }
72}
