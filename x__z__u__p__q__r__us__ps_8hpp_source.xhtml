89 inline constexpr void update(const UpdateTypes &...update_pack,
90 const auto &output_z, const auto &...outputs_z) {
91 update_arguments = {update_pack...};
92 z = output{output_z, outputs_z...};
93 s = h * p * t(h) + r;
94 k = p * t(h) / s;
95 y = z - h * x;
96 x = x + k * y;
97 p = (i - k * h) * p * t(i - k * h) + k * r * t(k);
98 }
99
101 inline constexpr void predict(const PredictionTypes &...prediction_pack,
102 const auto &input_u, const auto &...inputs_u) {
103 prediction_arguments = {prediction_pack...};
104 u = input{input_u, inputs_u...};
105 x = f * x + g * u;
106 p = f * p * t(f) + q;
107 }
108};
