277 inline constexpr void update(const UpdateArguments &...arguments,
278 const auto &...output_z)
279 {
280 const auto i{ identity.template operator()<estimate_uncertainty>() };
281
282 z = output{ output_z... };
283 h = observation_state_h(x, arguments...);
284 r = noise_observation_r(x, z, arguments...);
285 s = h * p * transpose(h) + r;
286 k = divide(p * transpose(h), s);
287 y = z - observation(x, arguments...);
288 x = x + k * y;
289 p = symmetrize(estimate_uncertainty{
290 (i - k * h) * p * transpose(i - k * h) + k * r * transpose(k) });
291 }
292
294 inline constexpr void predict(const PredictionArguments &...arguments,
295 const auto &...input_u)
296 {
297 u = input{ input_u... };
298 f = transition_state_f(x, arguments..., u);
299 q = noise_process_q(x, arguments...);
300 g = transition_control_g(arguments...);
301 x = f * x + g * u;
302 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
303 }
304
305 inline constexpr void predict(const PredictionArguments &...arguments)
306 {
307 f = transition_state_f(x, arguments..., input{});
308 q = noise_process_q(x, arguments...);
309 x = transition(x, arguments...);
310 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
311 }
