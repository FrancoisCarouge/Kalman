181 inline constexpr void update(const auto &...output_z)
182 {
183 h = transition_observation_h();
184 r = noise_observation_r();

186 internal::update<Transpose, Symmetrize, Divide, Identity>(x, p, h, r, z);

189 inline constexpr void predict(const PredictionArguments &...arguments,

192 // use member variables

198 internal::predict<Transpose, Symmetrize>(x, p, ff, f, q, g, u);
199 }
