201 inline constexpr void update(const auto &...output_z)
202 {
203 const auto z{ output{ output_z... } };
204 const auto i{ identity.template operator()<estimate_uncertainty>() };
205
206 h = transition_observation_h();
207 r = noise_observation_r();
208 s = h * p * transpose(h) + r;
209 k = divide(p * transpose(h), s);
210 y = z - h * x;
211 x = x + k * y;
212 p = symmetrize(estimate_uncertainty{
213 (i - k * h) * p * transpose(i - k * h) + k * r * transpose(k) });
214 }
215
218 inline constexpr void predict(const PredictionArguments &...arguments,
219 const auto &...input_u)
220 {
221 const auto u{ input{ input_u... } };
222
223 f = transition_state_f(arguments...);
224 q = noise_process_q(arguments...);
225 g = transition_control_g(arguments...);
226
227 x = f * x + g * u;
228 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
229 }
230
232 inline constexpr void predict(const PredictionArguments &...arguments)
233 {
234 f = transition_state_f(arguments...);
235 q = noise_process_q(arguments...);
236
237 x = f * x;
238 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
239 }
