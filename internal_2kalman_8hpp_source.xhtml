142 template <typename Output0, typename... OutputN>
143 inline constexpr void update(const UpdateTypes &...update_pack,
144 const Output0 &output_z,
145 const OutputN &...outputs_z) {
146 update_arguments = {update_pack...};
147 z = output{output_z, outputs_z...};
148 h = observation_state_h(x, update_pack...);
149 r = noise_observation_r(x, z, update_pack...);
150 s = innovation_uncertainty{h * p * t(h) + r};
151 k = p * t(h) / s;
152 y = z - observation(x, update_pack...);
153 x = state{x + k * y};
154 p = estimate_uncertainty{(i - k * h) * p * t(i - k * h) + k * r * t(k)};
155 }

157 inline constexpr void predict(const PredictionTypes &...prediction_pack) {
158 prediction_arguments = {prediction_pack...};
159 f = transition_state_f(x, prediction_pack...);
160 q = noise_process_q(x, prediction_pack...);
161 x = transition(x, prediction_pack...);
162 p = estimate_uncertainty{f * p * t(f) + q};
163 }
