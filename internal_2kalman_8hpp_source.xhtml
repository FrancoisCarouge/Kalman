191 inline constexpr void update(const auto &...output_z)
192 {
193 const auto z{ output{ output_z... } };
194
195 h = transition_observation_h();
196 r = noise_observation_r();
197
198 const innovation_uncertainty s{ h * p * transpose(h) + r };
199 const gain k{ divide(p * transpose(h), s) };
200 const innovation y{ z - h * x };
201 const auto i{ identity.template operator()<estimate_uncertainty>() };
202
203 x = state{ x + k * y };
204 p = symmetrize(estimate_uncertainty{
205 (i - k * h) * p * transpose(i - k * h) + k * r * transpose(k) });
206 }
207
208 inline constexpr void predict(const PredictionArguments &...arguments,
209 const auto &...input_u)
210 {
211 const auto ff{ predict_state };
212 const auto u{ input{ input_u... } };
213
214 f = transition_state_f(arguments...);
215 q = noise_process_q(arguments...);
216 g = transition_control_g(arguments...);
217
218 x = state{ ff(x, f) + g * u };
219 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
220 }
221
222 inline constexpr void predict(const PredictionArguments &...arguments)
223 {
224 const auto ff{ predict_state };
225
226 f = transition_state_f(arguments...);
227 q = noise_process_q(arguments...);
228
229 x = state{ ff(x, f) };
230 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
231 }
