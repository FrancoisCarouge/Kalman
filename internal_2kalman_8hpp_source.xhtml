70template <typename State, typename Output, typename Input, typename Transpose,
71 typename Symmetrize, typename Divide, typename Identity,
72 typename... UpdateTypes, typename... PredictionTypes>
73struct kalman<State, Output, Input, Transpose, Symmetrize, Divide, Identity,
74 pack<UpdateTypes...>, pack<PredictionTypes...>> {
75 using state = State;
76 using output = Output;
77 using input = Input;
78 using estimate_uncertainty =
79 std::decay_t<std::invoke_result_t<Divide, State, State>>;
80 using process_uncertainty =
81 std::decay_t<std::invoke_result_t<Divide, State, State>>;
82 using output_uncertainty =
83 std::decay_t<std::invoke_result_t<Divide, Output, Output>>;
84 using state_transition =
85 std::decay_t<std::invoke_result_t<Divide, State, State>>;
86 using output_model =
87 std::decay_t<std::invoke_result_t<Divide, Output, State>>;
88 using input_control =
89 std::decay_t<std::invoke_result_t<Divide, State, Input>>;
90 using gain = std::decay_t<std::invoke_result_t<Transpose, output_model>>;
91 using innovation = output;
92 using innovation_uncertainty = output_uncertainty;
93 using observation_state_function =
94 std::function<output_model(const state &, const UpdateTypes &...)>;
95 using noise_observation_function = std::function<output_uncertainty(
96 const state &, const output &, const UpdateTypes &...)>;
97 using transition_state_function = std::function<state_transition(
98 const state &, const PredictionTypes &..., const input &)>;
99 using noise_process_function = std::function<process_uncertainty(
100 const state &, const PredictionTypes &...)>;
101 using transition_control_function =
102 std::function<input_control(const PredictionTypes &...)>;
103 using transition_function =
104 std::function<state(const state &, const PredictionTypes &...)>;
105 using observation_function =
106 std::function<output(const state &, const UpdateTypes &...arguments)>;
107
109 state x{ 0 * Identity().template operator()<state>() };
110 estimate_uncertainty p{
111 Identity().template operator()<estimate_uncertainty>()
112 };
113 process_uncertainty q{
114 0 * Identity().template operator()<process_uncertainty>()
115 };
116 output_uncertainty r{ 0 *
117 Identity().template operator()<output_uncertainty>() };
118 output_model h{ Identity().template operator()<output_model>() };
119 state_transition f{ Identity().template operator()<state_transition>() };
120 input_control g{ Identity().template operator()<input_control>() };
121 gain k{ Identity().template operator()<gain>() };
122 innovation y{ 0 * Identity().template operator()<innovation>() };
123 innovation_uncertainty s{
124 Identity().template operator()<innovation_uncertainty>()
125 };
126 output z{ 0 * Identity().template operator()<output>() };
127 input u{ 0 * Identity().template operator()<input>() };
128
134 observation_state_function observation_state_h{
135 [this](const state &x, const UpdateTypes &...arguments) -> output_model {
136 static_cast<void>(x);
137 (static_cast<void>(arguments), ...);
138 return h;
139 }
140 };
141 noise_observation_function noise_observation_r{
142 [this](const state &x, const output &z,
143 const UpdateTypes &...arguments) -> output_uncertainty {
144 static_cast<void>(x);
145 static_cast<void>(z);
146 (static_cast<void>(arguments), ...);
147 return r;
148 }
149 };
150 transition_state_function transition_state_f{
151 [this](const state &x, const PredictionTypes &...arguments,
152 const input &u) -> state_transition {
153 static_cast<void>(x);
154 (static_cast<void>(arguments), ...);
155 static_cast<void>(u);
156 return f;
157 }
158 };
159 noise_process_function noise_process_q{
160 [this](const state &x,
161 const PredictionTypes &...arguments) -> process_uncertainty {
162 static_cast<void>(x);
163 (static_cast<void>(arguments), ...);
164 return q;
165 }
166 };
167 transition_control_function transition_control_g{
168 [this](const PredictionTypes &...arguments) -> input_control {
169 (static_cast<void>(arguments), ...);
170 return g;
171 }
172 };
173 transition_function transition{
174 [this](const state &x, const PredictionTypes &...arguments) -> state {
175 (static_cast<void>(arguments), ...);
176 return f * x;
177 }
178 };
179 observation_function observation{
180 [this](const state &x, const UpdateTypes &...arguments) -> output {
181 (static_cast<void>(arguments), ...);
182 return h * x;
183 }
184 };
185
186 Transpose transpose;
187 Divide divide;
188 Symmetrize symmetrize;
189 Identity identity;
190
195 inline constexpr void update(const UpdateTypes &...arguments,
196 const auto &...output_z)
197 {
198 const auto i{ identity.template operator()<estimate_uncertainty>() };
199
200 z = output{ output_z... };
201 h = observation_state_h(x, arguments...); // x, z, args?
202 r = noise_observation_r(x, z, arguments...);
203 s = h * p * transpose(h) + r;
204 k = divide(p * transpose(h), s);
205 y = z - observation(x, arguments...);
206 x = x + k * y;
207 p = symmetrize(estimate_uncertainty{
208 (i - k * h) * p * transpose(i - k * h) + k * r * transpose(k) });
209 }
210
217 inline constexpr void predict(const PredictionTypes &...arguments,
218 const auto &...input_u)
219 {
220 u = input{ input_u... };
221 f = transition_state_f(x, arguments..., u);
222 q = noise_process_q(x, arguments...);
223 g = transition_control_g(arguments...);
224 x = f * x + g * u;
225 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
226 }
227
228 inline constexpr void predict(const PredictionTypes &...arguments)
229 {
230 f = transition_state_f(x, arguments..., input{});
231 q = noise_process_q(x, arguments...);
232 x = transition(x, arguments...);
233 p = symmetrize(estimate_uncertainty{ f * p * transpose(f) + q });
234 }
235
236 inline constexpr void
237 operator()(const PredictionTypes &...prediction_arguments,
238 const UpdateTypes &...update_arguments, const auto &...input_u,
239 const auto &...output_z)
240 {
241 update(update_arguments..., output_z...);
242 predict(prediction_arguments..., input_u...);
243 }
244};
