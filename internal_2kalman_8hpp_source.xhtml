140 template <typename Output0, typename... OutputN>
141 inline constexpr void update(const UpdateTypes &...update_pack,
142 const Output0 &output_z,
143 const OutputN &...outputs_z) {
144 update_arguments = {update_pack...};
145 z = output{output_z, outputs_z...};
146 h = observation_state_h(x, update_pack...);
147 r = noise_observation_r(x, z, update_pack...);
148 s = innovation_uncertainty{h * p * t(h) + r};
149 k = p * t(h) / s;
150 y = z - observation(x, update_pack...);
151 x = state{x + k * y};
152 p = estimate_uncertainty{(identity_v<quotient<state, state>> - k * h) * p *
153 t(identity_v<quotient<state, state>> - k * h) +
154 k * r * t(k)};
155 }

157 inline constexpr void predict(const PredictionTypes &...prediction_pack) {
158 prediction_arguments = {prediction_pack...};
159 f = transition_state_f(x, prediction_pack...);
160 q = noise_process_q(x, prediction_pack...);
161 x = transition(x, prediction_pack...);
162 p = estimate_uncertainty{f * p * t(f) + q};
163 }
