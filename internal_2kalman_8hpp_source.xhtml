163 template <typename Output0, typename... OutputN>
164 inline constexpr void update(const UpdateTypes &...update_pack,
165 const Output0 &output_z,
166 const OutputN &...outputs_z) {
167 update_arguments = {update_pack...};
168 z = output{output_z, outputs_z...};
169 h = observation_state_h(x, update_pack...);
170 r = noise_observation_r(x, z, update_pack...);
171 s = innovation_uncertainty{h * p * t(h) + r};
172 k = divide(p * t(h), s);
173 y = z - observation(x, update_pack...);
174 x = state{x + k * y};
175 p = estimate_uncertainty{(i - k * h) * p * t(i - k * h) + k * r * t(k)};
176 }

178 inline constexpr void predict(const PredictionTypes &...prediction_pack) {
179 prediction_arguments = {prediction_pack...};
180 f = transition_state_f(x, prediction_pack...);
181 q = noise_process_q(x, prediction_pack...);
182 x = transition(x, prediction_pack...);
183 p = estimate_uncertainty{f * p * t(f) + q};
184 }
