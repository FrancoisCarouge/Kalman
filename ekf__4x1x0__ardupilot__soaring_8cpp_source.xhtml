70 const float measure_noise{std::pow(0.45f, 2.f)};

73 // Observation Z: [w] vertical air velocity w at the aircraft’s
74 // position w.r.t. the thermal center [m.s^-1].
75 k.observation([](const kalman::state &x, const float &position_x,
76 const float &position_y) -> kalman::output {
77 return x(0) * std::exp(-(std::pow(x[2] - position_x, 2.f) +
78 std::pow(x[3] - position_y, 2.f)) /
79 std::pow(x[1], 2.f));
80 });
81
82 // See the ArduSoar paper for the equation for H = ∂h/∂X:
83 k.h([](const kalman::state &x, const float &position_x,
84 const float &position_y) -> kalman::output_model {
85 const float expon{std::exp(
86 -(std::pow(x[2] - position_x, 2.f) + std::pow(x[3] - position_y, 2.f)) /
87 std::pow(x[1], 2.f))};
88
89 const kalman::output_model h{
90 expon,
91 2 * x(0) *
92 ((std::pow(x(2) - position_x, 2.f) +
93 std::pow(x(3) - position_y, 2.f)) /
94 std::pow(x(1), 3.f)) *
95 expon,
96 -2 * (x(0) * (x(2) - position_x) / std::pow(x(1), 2.f)) * expon,
97 -2 * (x(0) * (x(3) - position_y) / std::pow(x(1), 2.f)) * expon};
98
99 return h;
100 });
