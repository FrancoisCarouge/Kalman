242template <typename Type, std::size_t Row, std::size_t Column>
243[[nodiscard]] inline constexpr bool
244operator==(const matrix<Type, Row, Column> &lhs,
245 const matrix<Type, Row, Column> &rhs) {
246 for (decltype(Row) i{0}; i < Row; ++i) {
247 for (decltype(Column) j{0}; j < Column; ++j) {
248 if (lhs.data[i][j] != rhs.data[i][j]) {
249 return false;

257[[nodiscard]] inline constexpr auto

259 const matrix<Type, Size, Column> &rhs) {
260 matrix<Type, Row, Column> result;
261
262 for (decltype(Row) i{0}; i < Row; ++i) {
263 for (decltype(Column) j{0}; j < Column; ++j) {
