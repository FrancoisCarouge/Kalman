243template <typename Type, std::size_t Row, std::size_t Column>
244[[nodiscard]] inline constexpr bool
245operator==(const matrix<Type, Row, Column> &lhs,
246 const matrix<Type, Row, Column> &rhs) {
247 for (std::size_t i{0}; i < Row; ++i) {
248 for (std::size_t j{0}; j < Column; ++j) {

257template <typename Type, std::size_t Row, std::size_t Column, std::size_t Size>
258[[nodiscard]] inline constexpr auto
259operator*(const matrix<Type, Row, Size> &lhs,
260 const matrix<Type, Size, Column> &rhs) {
261 matrix<Type, Row, Column> result;
262
263 for (std::size_t i{0}; i < Row; ++i) {
264 for (std::size_t j{0}; j < Column; ++j) {
265 for (std::size_t k{0}; k < Size; ++k) {
266 result.data[i][j] += lhs.data[i][k] * rhs.data[k][j];

268 }
269 }

271 return result;

379 template <typename OutputIterator>
380 constexpr auto
381 format(const fcarouge::naive::matrix<Type, Row, Column> &value,
382 std::basic_format_context<OutputIterator, Char> &format_context) const
383 -> OutputIterator {
384 format_context.advance_to(std::format_to(format_context.out(), "["));
385
386 for (std::size_t i{0}; i < Row; ++i) {
387 if (i > 0) {
388 format_context.advance_to(std::format_to(format_context.out(), ", "));
389 }
390
391 format_context.advance_to(std::format_to(format_context.out(), "["));
392
393 for (std::size_t j{0}; j < Column; ++j) {
394 if (j > 0) {
395 format_context.advance_to(std::format_to(format_context.out(), ", "));
396 }
397
398 format_context.advance_to(
399 std::format_to(format_context.out(), "{}", value.data[i][j]));
400 }
401
402 format_context.advance_to(std::format_to(format_context.out(), "]"));
403 }

405 format_context.advance_to(std::format_to(format_context.out(), "]"));

407 return format_context.out();
