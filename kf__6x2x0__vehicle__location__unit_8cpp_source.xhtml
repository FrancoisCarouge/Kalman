84[[maybe_unused]] auto sample{[] {
85 // A 6x2x0 filter, constant acceleration dynamic model, no control.
86 kalman filter{
87 // The state X is chosen to be the position, velocity, acceleration in the
88 // XY plane: [px, vx, ax, py, vy, ay]. We don't know the vehicle location;
89 // we will set initial position, velocity and acceleration to 0.
90 state{0. * m, 0. * m / s, 0. * m / s2, 0. * m, 0. * m / s, 0. * m / s2},
91 // The vehicle has an onboard location sensor that reports output Z as X
92 // and Y coordinates of the system.
93 output<m, m>,
94 // The estimate uncertainty matrix P.
95 // Since our initial state vector is a guess, we will set a very high
96 // estimate uncertainty. The high estimate uncertainty results in a high
97 // Kalman Gain, giving a high weight to the measurement.
98 estimate_uncertainty{500. * one<estimate_uncertainty::type>},
99 // The process uncertainty noise matrix Q, constant, computed in place,
100 // with random acceleration standard deviation: σa = 0.2 m.s^-2.
101 process_uncertainty{[]() {
102 process_uncertainty::type value{one<process_uncertainty::type>};
103 value.at<0, 0>() = 0.25 * m2;
104 value.at<0, 1>() = 0.5 * m2 / s;
105 value.at<0, 2>() = 0.5 * m2 / s2;
106 value.at<1, 0>() = 0.5 * m2 / s;
107 value.at<1, 2>() = 1. * m2 / s3;
108 value.at<2, 0>() = 0.5 * m2 / s2;
109 value.at<2, 1>() = 1. * m2 / s3;
110 value.at<3, 3>() = 0.25 * m2;
111 value.at<3, 4>() = 0.5 * m2 / s;
112 value.at<3, 5>() = 0.5 * m2 / s2;
113 value.at<4, 3>() = 0.5 * m2 / s;
114 value.at<4, 5>() = 1. * m2 / s3;
115 value.at<5, 3>() = 0.5 * m2 / s2;
116 value.at<5, 4>() = 1. * m2 / s3;
117 return 0.2 * 0.2 * value;
118 }()},
119 // The output uncertainty matrix R. Assume that the x and y measurements
120 // are uncorrelated, i.e. error in the x coordinate measurement doesn't
121 // depend on the error in the y coordinate measurement. In real-life
122 // applications, the measurement uncertainty can differ between
123 // measurements. In many systems the measurement uncertainty depends on
124 // the measurement SNR (signal-to-noise ratio), angle between sensor (or
125 // sensors) and target, signal frequency and many other parameters.
126 // For the sake of the example simplicity, we will assume a constant
127 // measurement uncertainty: R1 = R2...Rn-1 = Rn = R The measurement error
128 // standard deviation: σxm = σym = 3m. The variance 9.
129 output_uncertainty{{9. * m2, 0. * m2}, {0. * m2, 9. * m2}},
130 // The output model matrix H. The dimension of zn is 2x1 and the dimension
131 // of xn is 6x1. Therefore the dimension of the observation matrix H shall
132 // be 2x6.
133 output_model{[]() {
134 output_model::type value{zero<output_model::type>};
135 value.at<0, 0>() = 1. * m2;
136 value.at<1, 3>() = 1. * m2;
137 return value;
138 }()},
139 // The state transition matrix F would be:
140 state_transition{[]() {
141 state_transition::type value{one<state_transition::type>};
142 value.at<0, 1>() = 1. * m2 / s;
143 value.at<0, 2>() = 0.5 * m2 / s2;
144 value.at<1, 2>() = 1. * m2 / s3;
145 value.at<3, 4>() = 1. * m2 / s;
146 value.at<3, 5>() = 0.5 * m2 / s2;
147 value.at<4, 5>() = 1. * m2 / s3;
148 return value;
149 }()}};
150
151 // Now we can predict the next state based on the initialization values.
152 filter.predict();
153
154 // The measurement values: z1 = [-393.66 m, 300.4 m]
155 filter.update(-393.66 * m, 300.4 * m);
156 filter.predict();
157 filter.update(-375.93 * m, 301.78 * m);
158 filter.predict();
159
160 // Verify the example estimated state at 0.1% accuracy.
161 assert(abs(1 - filter.x().at<0>() / (-277.8 * m)) < 0.001 &&
162 abs(1 - filter.x().at<1>() / (148.3 * m / s)) < 0.001 &&
163 abs(1 - filter.x().at<2>() / (94.5 * m / s2)) < 0.001 &&
164 abs(1 - filter.x().at<3>() / (249.8 * m)) < 0.001 &&
165 abs(1 - filter.x().at<4>() / (-85.9 * m / s)) < 0.001 &&
166 abs(1 - filter.x().at<5>() / (-63.62 * m / s2)) < 0.001 &&
167 "The state estimates expected at 0.1% accuracy.");
168
171
172 return 0;
173}()};
