70[[maybe_unused]] auto sample{[] {
71 // A 6x2x0 filter, constant acceleration dynamic model, no control.
72 kalman filter{
73 // The state X is chosen to be the position, velocity, acceleration in the
74 // XY plane: [px, vx, ax, py, vy, ay]. We don't know the vehicle location;
75 // we will set initial position, velocity and acceleration to 0.
76 state{state_t{position{0. * m}, velocity{0. * m / s},
77 acceleration{0. * m / s2}, position{0. * m},
78 velocity{0. * m / s}, acceleration{0. * m / s2}}},
79 // The vehicle has an onboard location sensor that reports output Z as X
80 // and Y coordinates of the system.
81 output<output_t>,
82 // The estimate uncertainty matrix P.
83 // Since our initial state vector is a guess, we will set a very high
84 // estimate uncertainty. The high estimate uncertainty results in a high
85 // Kalman Gain, giving a high weight to the measurement.
86 estimate_uncertainty{500. * one<ᴀʙᵀ<state_t, state_t>>},
87 // The process uncertainty noise matrix Q, constant, computed in place,
88 // with random acceleration standard deviation: σa = 0.2 m.s^-2.
89 process_uncertainty{[]() {
90 using process_uncertainty_t = ᴀʙᵀ<state_t, state_t>;
91 process_uncertainty_t value{one<process_uncertainty_t>};
92 value.at<0, 0>() = 0.25 * m2;
93 value.at<0, 1>() = 0.5 * m2 / s;
94 value.at<0, 2>() = 0.5 * m2 / s2;
95 value.at<1, 0>() = 0.5 * m2 / s;
96 value.at<1, 2>() = 1. * m2 / s3;
97 value.at<2, 0>() = 0.5 * m2 / s2;
98 value.at<2, 1>() = 1. * m2 / s3;
99 value.at<3, 3>() = 0.25 * m2;
100 value.at<3, 4>() = 0.5 * m2 / s;
101 value.at<3, 5>() = 0.5 * m2 / s2;
102 value.at<4, 3>() = 0.5 * m2 / s;
103 value.at<4, 5>() = 1. * m2 / s3;
104 value.at<5, 3>() = 0.5 * m2 / s2;
105 value.at<5, 4>() = 1. * m2 / s3;
106 return 0.2 * 0.2 * value;
107 }()},
108 // The output uncertainty matrix R. Assume that the x and y measurements
109 // are uncorrelated, i.e. error in the x coordinate measurement doesn't
110 // depend on the error in the y coordinate measurement. In real-life
111 // applications, the measurement uncertainty can differ between
112 // measurements. In many systems the measurement uncertainty depends on
113 // the measurement SNR (signal-to-noise ratio), angle between sensor (or
114 // sensors) and target, signal frequency and many other parameters.
115 // For the sake of the example simplicity, we will assume a constant
116 // measurement uncertainty: R1 = R2...Rn-1 = Rn = R The measurement error
117 // standard deviation: σxm = σym = 3m. The variance 9.
118 output_uncertainty{{9. * m2, 0. * m2}, {0. * m2, 9. * m2}},
119 // The output model matrix H. The dimension of zn is 2x1 and the dimension
120 // of xn is 6x1. Therefore the dimension of the observation matrix H shall
121 // be 2x6.
122 output_model{[]() {
123 using output_model_t = evaluate<quotient<output_t, state_t>>;
124 output_model_t value{zero<output_model_t>};
125 value.at<0, 0>() = 1. * m2;
126 value.at<1, 3>() = 1. * m2;
127 return value;
128 }()},
129 // The state transition matrix F would be:
130 state_transition{[]() {
131 using state_transition_t = evaluate<quotient<state_t, state_t>>;
132 state_transition_t value{one<state_transition_t>};
133 value.at<0, 1>() = 1. * m2 / s;
134 value.at<0, 2>() = 0.5 * m2 / s2;
135 value.at<1, 2>() = 1. * m2 / s3;
136 value.at<3, 4>() = 1. * m2 / s;
137 value.at<3, 5>() = 0.5 * m2 / s2;
138 value.at<4, 5>() = 1. * m2 / s3;

140 }()}};
