48 // Initialization
49 // The state is chosen to be the position, velocity, acceleration in the XY
50 // plane: [px, vx, ax, py, vy, ay]. We don't know the vehicle location; we
51 // will set initial position, velocity and acceleration to 0.
52 filter.x(0., 0., 0., 0., 0., 0.);
53
54 // Since our initial state vector is a guess, we will set a very high estimate
55 // uncertainty. The high estimate uncertainty results in a high Kalman Gain,
56 // giving a high weight to the measurement.
57 filter.p(kalman::estimate_uncertainty{{500, 0, 0, 0, 0, 0},
58 {0, 500, 0, 0, 0, 0},
59 {0, 0, 500, 0, 0, 0},
60 {0, 0, 0, 500, 0, 0},
61 {0, 0, 0, 0, 500, 0},
62 {0, 0, 0, 0, 0, 500}});
63
64 // Prediction
65 // The process noise matrix Q would be:
66 kalman::process_uncertainty q{
67 {0.25, 0.5, 0.5, 0, 0, 0}, {0.5, 1, 1, 0, 0, 0}, {0.5, 1, 1, 0, 0, 0},
68 {0, 0, 0, 0.25, 0.5, 0.5}, {0, 0, 0, 0.5, 1, 1}, {0, 0, 0, 0.5, 1, 1}};
69 q *= 0.2 * 0.2;

173 // when the vehicle starts the turning maneuver, the estimates are not so

175 // vehicle travels along the straight line, the acceleration is constant and
176 // equal to zero. However, during the turn maneuver, the vehicle experiences
177 // acceleration due to the circular motion - the angular acceleration.
178 // Although the angular acceleration is constant, the angular acceleration
