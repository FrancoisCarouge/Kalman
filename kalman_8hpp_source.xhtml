140 template <template <typename> typename Transpose,
141  template <typename> typename Identity>
142 [[nodiscard]] inline constexpr auto
143 update_covariance(const auto &p, const auto &k, const auto &h, const auto &r)
144 {
145  using estimate_uncertainty_p =
146  std::remove_reference_t<std::remove_cv_t<decltype(p)>>;
147  using gain = std::remove_reference_t<std::remove_cv_t<decltype(k)>>;
148  Transpose<estimate_uncertainty_p> transpose_p;
149  Transpose<gain> transpose_k;
150  Identity<estimate_uncertainty_p> i;
151 
152  return estimate_uncertainty_p{ (i() - k * h) * p * transpose_p(i() - k * h) +
153  k * r * transpose_k(k) };
154 }

156 template <template <typename> typename Transpose,
157  template <typename, typename> typename Divide>
158 [[nodiscard]] inline constexpr auto weight_gain(const auto &p, const auto &h,
159  const auto &r)
160 {
161  using observation_h = std::remove_reference_t<std::remove_cv_t<decltype(h)>>;
162  using measurement_uncertainty_r =
163  std::remove_reference_t<std::remove_cv_t<decltype(r)>>;
164  using gain = std::invoke_result_t<Transpose<observation_h>, observation_h>;

166  Divide<gain, measurement_uncertainty_r> divide;
167 
168  return gain{ divide(p * transpose_h(h), h * p * transpose_h(h) + r) };
