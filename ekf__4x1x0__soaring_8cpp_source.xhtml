43 // Initialization
44 const float trigger_strength{0};
45 const float thermal_radius{80};
46 const float thermal_position_x{5};
47 const float thermal_position_y{0};

51 const float strength_covariance{0.0049f};
52 const float radius_covariance{400};
53 const float position_covariance{400};

55 {0, radius_covariance, 0, 0},
56 {0, 0, position_covariance, 0},
57 {0, 0, 0, position_covariance}});

59 // No process dynamics: F = ∂f/∂X = I4 Default.
60
61 filter.transition([](const kalman::state &x, const float &drift_x,
62 const float &drift_y) -> kalman::state {
64 const kalman::state drifts{0, 0, drift_x, drift_y};
65 return x + drifts;
66 });
67
68 const float strength_noise{std::pow(0.001f, 2.f)};
69 const float distance_noise{std::pow(0.03f, 2.f)};
70 filter.q(kalman::process_uncertainty{{strength_noise, 0, 0, 0},
71 {0, distance_noise, 0, 0},
72 {0, 0, distance_noise, 0},
73 {0, 0, 0, distance_noise}});
74
75 const float measure_noise{std::pow(0.45f, 2.f)};
76 filter.r(kalman::output_uncertainty{measure_noise});
77
78 // Observation Z: [w] vertical air velocity w at the aircraft’s
79 // position w.r.t. the thermal center [m.s^-1].
80 filter.observation([](const kalman::state &x, const float &position_x,
81 const float &position_y) -> kalman::output {
82 return kalman::output{x(0) * std::exp(-(std::pow(x[2] - position_x, 2.f) +
83 std::pow(x[3] - position_y, 2.f)) /
84 std::pow(x[1], 2.f))};
85 });
86
87 // See the ArduSoar paper for the equation for H = ∂h/∂X:
88 filter.h([](const kalman::state &x, const float &position_x,
89 const float &position_y) -> kalman::output_model {
90 const float expon{std::exp(
91 -(std::pow(x[2] - position_x, 2.f) + std::pow(x[3] - position_y, 2.f)) /
92 std::pow(x[1], 2.f))};
93
94 const kalman::output_model h{
95 expon,
96 2 * x(0) *
97 ((std::pow(x(2) - position_x, 2.f) +
98 std::pow(x(3) - position_y, 2.f)) /
99 std::pow(x(1), 3.f)) *
100 expon,
101 -2 * (x(0) * (x(2) - position_x) / std::pow(x(1), 2.f)) * expon,
102 -2 * (x(0) * (x(3) - position_y) / std::pow(x(1), 2.f)) * expon};
103
104 return h;
105 });
