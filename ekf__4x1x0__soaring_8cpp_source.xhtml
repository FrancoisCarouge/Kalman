84 std::pow(x[3] - position_y, 2.F)) /
85 std::pow(x[1], 2.F))};
86 });

88 // See the ArduSoar paper for the equation for H = ∂h/∂X:
89 filter.h([](const kalman::state &x, const float &position_x,
90 const float &position_y) -> kalman::output_model {
91 const float expon{std::exp(
92 -(std::pow(x[2] - position_x, 2.F) + std::pow(x[3] - position_y, 2.F)) /
93 std::pow(x[1], 2.F))};
94
95 const kalman::output_model h{
96 expon,
97 2 * x(0) *
98 ((std::pow(x(2) - position_x, 2.F) +
99 std::pow(x(3) - position_y, 2.F)) /
100 std::pow(x(1), 3.F)) *
101 expon,
102 -2 * (x(0) * (x(2) - position_x) / std::pow(x(1), 2.F)) * expon,
103 -2 * (x(0) * (x(3) - position_y) / std::pow(x(1), 2.F)) * expon};
104
105 return h;
106 });
