549template <std::size_t Begin, std::size_t End, std::size_t Increment,
550 typename Function>
551inline constexpr void for_constexpr(Function &&function) {
552 if constexpr (Begin < End) {
553 function(std::integral_constant<std::size_t, Begin>());
554 for_constexpr<Begin + Increment, End, Increment>(
555 std::forward<Function>(function));
556 }
