359 template <typename OutputIterator>
360 constexpr auto format(
361 const fcarouge::indexed::matrix<Matrix, RowIndexes, ColumnIndexes> &value,
362 std::basic_format_context<OutputIterator, Char> &format_context) const
363 -> OutputIterator {
364 format_context.advance_to(std::format_to(format_context.out(), "["));
365
366 for (std::size_t i{0}; i < fcarouge::size<RowIndexes>; ++i) {
367 if (i > 0) {
368 format_context.advance_to(std::format_to(format_context.out(), ", "));
369 }
370
371 format_context.advance_to(std::format_to(format_context.out(), "["));
372
373 for (std::size_t j{0}; j < fcarouge::size<ColumnIndexes>; ++j) {
374 if (j > 0) {
375 format_context.advance_to(std::format_to(format_context.out(), ", "));
376 }
377
378 format_context.advance_to(
379 std::format_to(format_context.out(), "{}", value.data(i, j)));
380 }
381
382 format_context.advance_to(std::format_to(format_context.out(), "]"));
383 }
384
385 format_context.advance_to(std::format_to(format_context.out(), "]"));
386
387 return format_context.out();
388 }
