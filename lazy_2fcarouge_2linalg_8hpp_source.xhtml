321template <typename Type, auto Row, auto Column, bool CopyableOrNot1,
322 bool CopyableOrNot2>
323[[nodiscard]] inline constexpr bool
324operator==(matrix<Type, Row, Column, CopyableOrNot1> lhs,
325 matrix<Type, Row, Column, CopyableOrNot2> rhs) {
326 std::array<Type, Row * Column> lhs_elements; // std::ranges::to
327 std::ranges::copy(lhs.genie, lhs_elements.begin());
328 lhs.genie = make_generator(lhs_elements);
329
330 std::array<Type, Row * Column> rhs_elements; // std::ranges::to
331 std::ranges::copy(rhs.genie, rhs_elements.begin());
332 rhs.genie = make_generator(rhs_elements);
333
334 return lhs_elements == rhs_elements;
335}
