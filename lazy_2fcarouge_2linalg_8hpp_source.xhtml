65template <std::ranges::range Range>
66inline constexpr std::generator<typename Range::value_type>
67make_generator(Range elements) {
68 return [](auto elements_copy)
69 -> std::generator<typename decltype(elements)::value_type> {
70 for (auto &&element : elements_copy) { // std::ranges::elements_of
71 co_yield element;
72 }

306template <typename Type, auto Row, auto Column>
307auto make_identity_generator{[]() -> std::generator<Type> {
308 for (decltype(Row) i{0}; i < Row; ++i) {
309 for (decltype(Column) j{0}; j < Column; ++j) {
310 co_yield i == j;
311 }
312 }
313}};

357template <typename Type, auto Row, auto Column, bool CopyableOrNot1,
358 bool CopyableOrNot2>
359[[nodiscard]] inline constexpr bool
360operator==(matrix<Type, Row, Column, CopyableOrNot1> lhs,
361 matrix<Type, Row, Column, CopyableOrNot2> rhs) {
362 std::array<Type, Row * Column> lhs_elements; // std::ranges::to
363 std::ranges::copy(lhs.genie, lhs_elements.begin());
364 lhs.genie = make_generator(lhs_elements);
365
366 std::array<Type, Row * Column> rhs_elements; // std::ranges::to
367 std::ranges::copy(rhs.genie, rhs_elements.begin());
368 rhs.genie = make_generator(rhs_elements);
369
370 return lhs_elements == rhs_elements;
371}
