71[[maybe_unused]] auto sample{[] {
72 kalman filter{
73 // The six estimated states X initialization under the simulated scenario:
74 // the lunar module is 30km away, approaching at 100 m/s relative to
75 // command/service module.
76 // Position: x=30km, y=1km, z=0.5km.
77 // Velocity: Closing speed 100m/s.
78 state{30000., 1000., 500., -100., 0., 0.},
79 // The measurement definition Z for the Rendezvous Radar: range (r),
80 // range rate (r_dot), shaft angle (beta), trunnion angle (theta).
81 output<vector<4>>,
82 // The initial estimate uncertainty P is a high uncertainty in position
83 // before radar lock. The velocity is somewhat known.
84 estimate_uncertainty{{1000., 0., 0., 0., 0., 0.},
85 {0., 1000., 0., 0., 0., 0.},
86 {0., 0., 1000., 0., 0., 0.},
87 {0., 0., 0., 100., 0., 0.},
88 {0., 0., 0., 0., 100., 0.},
89 {0., 0., 0., 0., 0., 100.}},
90 // The process uncertainty Q is a small accelerometer noise/drift.
91 process_uncertainty{{0.01, 0., 0., 0., 0., 0.},
92 {0., 0.01, 0., 0., 0., 0.},
93 {0., 0., 0.01, 0., 0., 0.},
94 {0., 0., 0., 0.01, 0., 0.},
95 {0., 0., 0., 0., 0.01, 0.},
96 {0., 0., 0., 0., 0., 0.01}},
97 // The output uncertainty R is the radar specifications.
98 // Range error: ~30m, Rate error: ~0.5m/s, Angles error: ~0.005 rad.
99 output_uncertainty{{30. * 30., 0., 0., 0.},
100 {0., 0.5 * 0.5, 0., 0.},
101 {0., 0., 0.005 * 0.005, 0.},
102 {0., 0., 0., 0.005 * 0.005}},
103 // The output model H is the Jacobian, linearization around the current
104 // state. Complex derivation omitted for brevity, using a simplified
105 // approximation.
106 output_model{[](const vector<6> &x) {
107 const auto &[rx, ry, rz, vx, vy, vz]{x};
108 double range{std::sqrt(rx * rx + ry * ry + rz * rz)};
109 // For production, guard against divide by zero range, if the spacecraft
110 // are touching.
111 double range_rate{(rx * vx + ry * vy + rz * vz) / range};
112 matrix<4, 6> h{matrix<4, 6>::Zero()};
113
114 // For range and range rate: the quotient rule applications for ∂ṙ/∂x.
115 // d(Range) / d(State)
116 h[0, 0] = rx / range;
117 h[0, 1] = ry / range;
118 h[0, 2] = rz / range;
119
120 // d(RangeRate) / d(State): Complex derivation omitted for brevity,
121 // simplified approximation.
122 h[1, 0] = (vx * range - rx * range_rate) / (range * range); // dRR/dx
123 h[1, 1] = (vy * range - ry * range_rate) / (range * range); // dRR/dy
124 h[1, 2] = (vz * range - rz * range_rate) / (range * range); // dRR/dz
125 h[1, 3] = rx / range; // dRR/dvx
126 h[1, 4] = ry / range; // dRR/dvy
127 h[1, 5] = rz / range; // dRR/dvz
128
129 // For the shaft/azimuth: the derivative of atan2.
130 // d(Shaft) / d(State)
131 double r2_xy{rx * rx + ry * ry};
132 // For production, guard against divide by zero.
133 h[2, 0] = -ry / r2_xy;
134 h[2, 1] = rx / r2_xy;
135
136 // For the trunnion/elevation: the derivative of asin(z/r):
137 // d(z/r)/dState * sqrt(1-(z/r)^2). d(Trunnion) / d(State)
138 double term{std::sqrt(1 - (rz * rz) / (range * range))};
139 h[3, 0] = (-rz * rx) / (range * range * range * term);
140 h[3, 1] = (-rz * ry) / (range * range * range * term);
141 h[3, 2] = (range * range - rz * rz) / (range * range * range * term);
142
143 return h;
144 }},
145 // The state transition matrix F:
146 state_transition{
147 []([[maybe_unused]] const vector<6> &x, const double &dt) {
148 return matrix<6, 6>{{1., 0., 0., dt, 0., 0.}, //
149 {0., 1., 0., 0., dt, 0.}, //
150 {0., 0., 1., 0., 0., dt}, //
151 {0., 0., 0., 1., 0., 0.}, //
152 {0., 0., 0., 0., 1., 0.}, //
153 {0., 0., 0., 0., 0., 1.}};
154 }},
155 // The observation estimation Z:
156 observation{[](const vector<6> &x) {
157 const auto &[rx, ry, rz, vx, vy, vz]{x};
158 double range{std::sqrt(rx * rx + ry * ry + rz * rz)};
159 // For production, guard against divide by zero.
160 double range_rate{(rx * vx + ry * vy + rz * vz) / range};
161 // The shaft angle in the xy plane usually, or defined by radar gimbal
162 // is simplified here for illustration.
163 double shaft{std::atan2(ry, rx)};
164 double trunnion{std::asin(rz / range)};
165
166 return vector<4>{range, range_rate, shaft, trunnion};
167 }},
168 // One additional parameter for prediction: time update (dt).
169 prediction_types<double>
170 // For production, the innovation angles, residuals should be normalized,
171 // modulo in [-Pi; +Pi], to avoid failures at the wrap boundary.
172 };
173
174 // In a real system, the hardware would provide the data. True state would be
175 // approximately `pos + vel*dt`. Simulate receiving 10 radar measurements, one
176 // per second. Fake data decreasing range by 100m each second, with slight
177 // angle offsets.
178 for (int k = 0; k < 10; ++k) {
179 filter.predict(1.);
180 double true_range{30000. - (100. * (k + 1))};
181 filter.update(true_range, -100., 0.03, 0.01);
182 }
183
184 double range{vector<3>{filter.x().template head<3>()}.norm()};
185 double velocity{filter.x()[3]};
186
187 assert(std::abs(1 - range / 29'001.861'093'990) < 0.000'000'001 &&
188 std::abs(1 - velocity / -99.574'527'631'012) < 0.000'000'001 &&
189 "After simulating 10 seconds, the estimated range is 29,001.86 meters "
190 "and an estimated closing velocity of 99.57 m/s.");
191
192 return 0;
193}()};
