111 template <template <typename> typename Transpose,
112  template <typename> typename Identity>
113 [[nodiscard]] inline constexpr auto
114 update_covariance(const auto &p, const auto &k, const auto &h, const auto &r)
115 {
116  using estimate_uncertainty_p =
117  std::remove_reference_t<std::remove_cv_t<decltype(p)>>;
118  using gain = std::remove_reference_t<std::remove_cv_t<decltype(k)>>;
119  Transpose<estimate_uncertainty_p> transpose_p;
120  Transpose<gain> transpose_k;
121  Identity<estimate_uncertainty_p> i;
122 
123  return estimate_uncertainty_p{ (i() - k * h) * p * transpose_p(i() - k * h) +
124  k * r * transpose_k(k) };
125 }

127 template <template <typename> typename Transpose,
128  template <typename, typename> typename Divide>
129 [[nodiscard]] inline constexpr auto weight_gain(const auto &p, const auto &h,
130  const auto &r)
131 {
132  using observation = std::remove_reference_t<std::remove_cv_t<decltype(h)>>;
133  using observation_noise_uncertainty =
134  std::remove_reference_t<std::remove_cv_t<decltype(r)>>;
135  using gain = std::invoke_result_t<Transpose<observation>, observation>;
136  Transpose<observation> transpose_h;
137  Divide<gain, observation_noise_uncertainty> divide;
138 
139  return gain{ divide(p * transpose_h(h), h * p * transpose_h(h) + r) };
140 }
